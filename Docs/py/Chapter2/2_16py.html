VΑΒ=I12*R2-I34*R4

print('Έστω R5 η αντίσταση που θα τοποθετήσουμε παράλληλα στην R4')
print('Έστω I1 το ρεύμα που διέρχεται από την R1 και R2')
print('Έστω I3 το ρεύμα που διέρχεται από την R3')

I1 = V / (R1+R2)

I4 = I1*R2 / R4
